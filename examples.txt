lambda n : Nat. if iszero n then 0 else if iszero (pred n) then 1 else sum (fib (pred (pred n))) (fib (pred n)) in
fib

fact = letrec fact : Nat -> Nat =
lambda n : Nat. if iszero n then 1 else prod n (fact (pred n)) in
fact
